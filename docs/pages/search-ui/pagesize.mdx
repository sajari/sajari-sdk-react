# PageSize

The PageSize component renders a select component for page sizes.

```js
import { Input, Results, PageSize } from '@sajari/react-search-ui';
import { SearchContextProvider, Pipeline } from '@sajari/react-hooks';
```

## Usage

```jsx
function Example() {
  const pipeline = new Pipeline(
    {
      project: '1594153711901724220',
      collection: 'bestbuy',
      endpoint: 'https://jsonapi-us-valkyrie.sajari.net',
    },
    'query',
  );

  const SearchPlayground = React.memo(() => (
    <div className="space-y-4">
      <div className="flex items-center space-x-4">
        <div className="flex-1">
          <Input />
        </div>

        <PageSize />
      </div>

      <Results />

      <div className="text-center">
        <Pagination />
      </div>
    </div>
  ));

  return (
    <SearchContextProvider search={{ pipeline, fields: { title: 'name', subtitle: 'brand' } }} searchOnLoad>
      <SearchPlayground />
    </SearchContextProvider>
  );
}
```

## Props

| Name    | Type                 | Default               | Description                   |
| ------- | -------------------- | --------------------- | ----------------------------- |
| `label` | `string`             | `Show`                |                               |
| `sizes` | `Array<number>`      | [10, 20, 35, 50, 100] |                               |
| `size`  | `sm` \| `md` \| `lg` | `md`                  | Sets the text and input size. |
